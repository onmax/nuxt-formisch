# Customization

AutoForm provides multiple layers of customization: per-field config, global theme overrides, and custom components.

## Field Config

The `fieldConfig` prop overrides field behavior without modifying your schema:

```vue [app.vue]
<template>
  <AutoForm
    :schema="schema"
    :field-config="fieldConfig"
    @submit="onSubmit"
  >
    <template #actions>
      <UButton type="submit">Save</UButton>
    </template>
  </AutoForm>
</template>

<script setup lang="ts">
import * as v from 'valibot'

const schema = v.object({
  email: v.pipe(v.string(), v.email()),
  password: v.string(),
  role: v.picklist(['admin', 'user']),
  internal_notes: v.optional(v.string())
})

const fieldConfig = {
  email: {
    label: 'Email Address',
    placeholder: 'you@example.com',
    description: 'We will never share your email'
  },
  password: {
    label: 'Password',
    section: 'Security'
  },
  role: {
    section: 'Security',
    order: 1
  },
  internal_notes: {
    hidden: true
  }
}

function onSubmit(data: v.InferOutput<typeof schema>) {
  console.log(data)
}
</script>
```

### Field Config Options

| Option | Type | Description |
|--------|------|-------------|
| `label` | `string` | Override field label |
| `placeholder` | `string` | Input placeholder text |
| `description` | `string` | Help text below the field |
| `unit` | `string` | Unit suffix (e.g., "$", "kg") |
| `section` | `string` | Group under section heading |
| `order` | `number` | Sort order within section |
| `hidden` | `boolean` | Hide field from form |
| `colSpan` | `1-4` | Grid column span |
| `disabled` | `boolean` | Disable this field |
| `component` | `Component` | Custom Vue component |
| `componentProps` | `Record<string, unknown>` | Props for custom component |

## Theme

AutoForm uses Tailwind Variants for styling. Override defaults in `app.config.ts`:

```ts [app.config.ts]
export default defineAppConfig({
  ui: {
    autoForm: {
      root: 'space-y-8',
      section: 'border-b pb-6',
      sectionTitle: 'text-xl font-bold mb-4 text-primary',
      grid: 'grid gap-6'
    }
  }
})
```

### Available Slots

| Slot | Default | Description |
|------|---------|-------------|
| `root` | `space-y-6` | Form wrapper |
| `section` | `''` | Section container |
| `sectionTitle` | `text-lg font-medium mb-3` | Section heading |
| `grid` | `grid gap-4 grid-cols-{columns}` | Field grid |

### Instance Override

Override styles on a single instance with the `ui` prop:

```vue [app.vue]
<template>
  <AutoForm
    :schema="schema"
    :ui="{
      root: 'space-y-4',
      grid: 'grid gap-2'
    }"
    @submit="onSubmit"
  >
    <template #actions>
      <UButton type="submit">Save</UButton>
    </template>
  </AutoForm>
</template>
```

## Custom Components

Replace the default input for a field with any Vue component:

```vue [app.vue]
<template>
  <AutoForm
    :schema="schema"
    :field-config="fieldConfig"
    @submit="onSubmit"
  >
    <template #actions>
      <UButton type="submit">Save</UButton>
    </template>
  </AutoForm>
</template>

<script setup lang="ts">
import * as v from 'valibot'
import RichTextEditor from '~/components/RichTextEditor.vue'
import ColorPicker from '~/components/ColorPicker.vue'

const schema = v.object({
  title: v.string(),
  content: v.string(),
  accentColor: v.string()
})

const fieldConfig = {
  content: {
    component: RichTextEditor,
    componentProps: {
      toolbar: ['bold', 'italic', 'link']
    }
  },
  accentColor: {
    component: ColorPicker,
    componentProps: {
      swatches: ['#ff0000', '#00ff00', '#0000ff']
    }
  }
}

function onSubmit(data: v.InferOutput<typeof schema>) {
  console.log(data)
}
</script>
```

### Custom Component Contract

Custom components receive these props:

| Prop | Type | Description |
|------|------|-------------|
| `modelValue` | `unknown` | Current field value |
| `field` | `ResolvedField` | Field metadata |
| `error` | `string \| undefined` | Validation error |
| `disabled` | `boolean` | Disabled state |

Emit `update:modelValue` to sync the value:

```vue [components/ColorPicker.vue]
<template>
  <UFormField :label="field.ui.label" :error="error">
    <div class="flex gap-2">
      <button
        v-for="color in swatches"
        :key="color"
        :style="{ backgroundColor: color }"
        :class="{ 'ring-2': modelValue === color }"
        class="size-8 rounded"
        :disabled="disabled"
        @click="$emit('update:modelValue', color)"
      />
    </div>
  </UFormField>
</template>

<script setup lang="ts">
import type { ResolvedField } from 'nuxt-formisch'

defineProps<{
  modelValue: string
  field: ResolvedField
  error?: string
  disabled?: boolean
  swatches: string[]
}>()

defineEmits<{
  'update:modelValue': [value: string]
}>()
</script>
```

## Combining Approaches

Use schema metadata for reusable defaults, fieldConfig for instance-specific overrides:

```ts [shared/utils/schemas.ts]
import * as v from 'valibot'

export const userSchema = v.object({
  email: v.pipe(
    v.string(),
    v.email(),
    v.title('Email'),
    v.description('Your primary email address')
  ),
  role: v.pipe(
    v.picklist(['admin', 'user', 'guest']),
    v.title('Role')
  )
})
```

```vue [pages/admin/users.vue]
<template>
  <AutoForm
    :schema="userSchema"
    :field-config="{
      role: { disabled: !isAdmin }
    }"
    @submit="onSubmit"
  >
    <template #actions>
      <UButton type="submit">Save</UButton>
    </template>
  </AutoForm>
</template>

<script setup lang="ts">
import { userSchema } from '#shared/utils/schemas'

const isAdmin = useAuth().user.role === 'admin'

function onSubmit(data: v.InferOutput<typeof userSchema>) {
  console.log(data)
}
</script>
```
